# 开发阶段 - Phase 3: 端云协作仲裁器

> 解决技术风险：Token 破产、Lua 脱缰野马

---

## 阶段目标

1. 实现意图分流路由器，90% 指令本地执行
2. 实现事件清洗层，日事件量从 86400 降至 <100
3. 实现 Lua 沙盒安全机制，防止大模型幻觉导致的系统崩溃
4. 实现优先级仲裁器，解决端云决策冲突

---

## 技术风险应对

### 风险二：Token 破产

| 问题 | 解决方案 | 预期收益 |
|------|---------|---------|
| 事件风暴 86400/天 | 状态突变上报 | 削减 99.9% |
| 持续上报消耗 | 伴随式上报 | 按需触发 |

### 风险三：Lua 脱缰野马

| 问题 | 解决方案 | 预期收益 |
|------|---------|---------|
| 大模型生成死循环 | 静态分析 + Watchdog | 强制终止 |
| 高频硬件调用 | 限流器 | 保护硬件 |

// 高频命令词表
typedef struct {
    const char* command;
    void (*action)(void);
    uint32_t hit_count;
} command_entry_t;

static command_entry_t command_table[] = {
    {"开灯",     action_light_on,      0},
    {"关灯",     action_light_off,     0},
    {"亮一点",   action_brighten,      0},
    {"暗一点",   action_dim,           0},
    {"冷光",     action_cool_light,    0},
    {"暖光",     action_warm_light,    0},
    {"自动",     action_auto_mode,     0},
};

#define COMMAND_TABLE_SIZE (sizeof(command_table) / sizeof(command_entry_t))

// 精确匹配查找
command_entry_t* find_command(const char* text) {
    for (int i = 0; i < COMMAND_TABLE_SIZE; i++) {
        if (strcmp(text, command_table[i].command) == 0) {
            command_table[i].hit_count++;
            return &command_table[i];
        }
    }
    return NULL;
}

**状态突变检测代码：**

```c
typedef enum {
    PRESENCE_NONE,
    PRESENCE_DETECTED,
    PRESENCE_CONFIRMED,
} presence_state_t;

typedef struct {
    presence_state_t state;
    int64_t state_change_time;
    int64_t confirm_duration_ms;  // 3 分钟 = 180000ms
} presence_detector_t;

bool check_presence_change(presence_detector_t* det, bool radar_detected) {
    int64_t now = esp_timer_get_time() / 1000;  // ms
    
    if (radar_detected && det->state == PRESENCE_NONE) {
        // 检测到有人，开始确认计时
        det->state = PRESENCE_DETECTED;
        det->state_change_time = now;
        return false;  // 不上报，等待确认
    }
    
    if (radar_detected && det->state == PRESENCE_DETECTED) {
        // 持续检测，检查是否达到确认时间
        if (now - det->state_change_time > det->confirm_duration_ms) {
            det->state = PRESENCE_CONFIRMED;
            return true;  // 确认状态突变，上报
        }
    }
    
    if (!radar_detected && det->state != PRESENCE_NONE) {
        // 检测不到人，状态回滚
        det->state = PRESENCE_NONE;
        det->state_change_time = now;
        return false;
    }
    
    return false;
}
```

**硬件调用限流器代码：**

```c
typedef struct {
    const char* func_name;
    int64_t window_ms;      // 时间窗口
    int max_calls;          // 最大调用次数
    int call_count;         // 当前计数
    int64_t window_start;   // 窗口起始时间
} rate_limiter_t;

static rate_limiter_t limiters[] = {
    {"set_lighting", 10000, 2, 0, 0},   // 10秒内最多2次
    {"send_notify",  60000, 1, 0, 0},   // 60秒内最多1次
};

bool check_rate_limit(const char* func_name) {
    int64_t now = esp_timer_get_time() / 1000;  // ms
    
    for (int i = 0; i < sizeof(limiters)/sizeof(rate_limiter_t); i++) {
        if (strcmp(func_name, limiters[i].func_name) == 0) {
            // 检查是否需要重置窗口
            if (now - limiters[i].window_start > limiters[i].window_ms) {
                limiters[i].window_start = now;
                limiters[i].call_count = 0;
            }
            
            // 检查是否超限
            if (limiters[i].call_count >= limiters[i].max_calls) {
                ESP_LOGW(TAG, "Rate limit exceeded: %s", func_name);
                return false;  // 拒绝调用
            }
            
            limiters[i].call_count++;
            return true;
        }
    }
    
    return true;  // 未配置限流器的函数，允许调用
}
```
