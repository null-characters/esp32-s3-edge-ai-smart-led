## 附录: 关键数据结构

### sound_features.h
```c
#ifndef SOUND_FEATURES_H
#define SOUND_FEATURES_H

#define MFCC_NUM 40
#define MFCC_FRAMES 32

typedef struct {
    float mfcc[MFCC_NUM][MFCC_FRAMES];  // MFCC 特征
    float vad;                           // VAD 状态
} sound_features_t;

int sound_init(void);
int sound_read(sound_features_t *features);
int sound_infer(const sound_features_t *features, float *probs);

#endif
```
