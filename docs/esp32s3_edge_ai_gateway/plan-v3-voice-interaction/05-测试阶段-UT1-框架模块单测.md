## 测试脚本

```c
// test_framework_module.c
#include "unity.h"
#include "led_pwm.h"
#include "uart_driver.h"
#include "i2s_driver.h"
#include "wifi_sta.h"

// LED PWM 测试
TEST_CASE("LED PWM 亮度 0%", "[framework]") {
    led_set_brightness(0);
    TEST_ASSERT_EQUAL(0, led_get_duty());
}

TEST_CASE("LED PWM 亮度 50%", "[framework]") {
    led_set_brightness(50);
    TEST_ASSERT_EQUAL(50, led_get_duty());
}

TEST_CASE("LED PWM 亮度 100%", "[framework]") {
    led_set_brightness(100);
    TEST_ASSERT_EQUAL(100, led_get_duty());
}

// I2S 麦克风测试
TEST_CASE("I2S 麦克风采样", "[framework]") {
    int16_t audio_buffer[1024];
    size_t bytes_read = i2s_read_audio(audio_buffer, 1024);
    TEST_ASSERT_EQUAL(1024 * 2, bytes_read);
}

// UART 通信测试
TEST_CASE("UART 数据帧解析", "[framework]") {
    uint8_t test_frame[] = {0xF1, 0xAA, 0x55, 0x01, 0x02, 0xF2};
    radar_data_t data;
    TEST_ASSERT_EQUAL(ESP_OK, uart_parse_frame(test_frame, &data));
}

// Wi-Fi 连接测试
TEST_CASE("Wi-Fi STA 连接", "[framework]") {
    TEST_ASSERT_EQUAL(ESP_OK, wifi_sta_connect());
    TEST_ASSERT_TRUE(wifi_is_connected());
}
```
