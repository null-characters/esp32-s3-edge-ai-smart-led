## 测试脚本

```c
// test_arbitration_module.c
#include "unity.h"
#include "priority_arbiter.h"
#include "ttl_lease.h"
#include "env_watcher.h"

// TTL 测试
TEST_CASE("TTL过期释放", "[arbitration]") {
    arbiter_set_mode(MODE_VOICE_LOCK);
    arbiter_set_ttl(100);  // 100 秒
    
    // 模拟时间流逝
    for (int i = 0; i < 100; i++) {
        ttl_lease_tick();
    }
    
    TEST_ASSERT_EQUAL(MODE_AUTO, arbiter_get_mode());
}

// 环境变化测试
TEST_CASE("环境变化释放", "[arbitration]") {
    arbiter_set_mode(MODE_VOICE_LOCK);
    
    // 模拟人离开 10 分钟
    env_watcher_set_presence(false);
    for (int i = 0; i < 600; i++) {  // 600 秒
        env_watcher_tick();
    }
    
    TEST_ASSERT_EQUAL(MODE_AUTO, arbiter_get_mode());
}
```
