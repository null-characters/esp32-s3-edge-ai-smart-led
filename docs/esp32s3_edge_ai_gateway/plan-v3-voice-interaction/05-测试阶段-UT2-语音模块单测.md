## 测试脚本

```c
// test_voice_module.c
#include "unity.h"
#include "wake_word.h"
#include "command_handler.h"
#include "tts_engine.h"
#include "aec_pipeline.h"
#include "vad_detector.h"

// 唤醒词测试
TEST_CASE("唤醒词识别率", "[voice]") {
    int success = 0;
    for (int i = 0; i < 100; i++) {
        if (wake_word_detect() == WAKE_WORD_DETECTED) {
            success++;
        }
    }
    TEST_ASSERT_GREATER_THAN(95, success);  // > 95%
}

// 命令词测试
TEST_CASE("命令词识别率", "[voice]") {
    int success = 0;
    for (int i = 0; i < 500; i++) {
        voice_cmd_t cmd = command_recognize();
        if (cmd.type != CMD_UNKNOWN) {
            success++;
        }
    }
    TEST_ASSERT_GREATER_THAN(450, success);  // > 90%
}

// AEC 测试
TEST_CASE("AEC 回声消除", "[voice]") {
    // 播放提示音
    tts_play_prompt("好的，主人");
    
    // 同时检测唤醒词
    vTaskDelay(pdMS_TO_TICKS(100));  // 等待播放
    
    // 不应该误触发
    TEST_ASSERT_NOT_EQUAL(WAKE_WORD_DETECTED, wake_word_detect());
}

// VAD 动态路由测试
TEST_CASE("VAD 动态路由", "[voice]") {
    // 无人声时
    TEST_ASSERT_EQUAL(ROUTE_TO_TFLM, audio_router_get_state());
    
    // 模拟人声
    vad_set_voice_detected(true);
    vTaskDelay(pdMS_TO_TICKS(50));
    
    // 应切换到 ESP-SR
    TEST_ASSERT_EQUAL(ROUTE_TO_ESP_SR, audio_router_get_state());
}
```
