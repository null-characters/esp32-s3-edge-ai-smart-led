## 附录: 场景识别接口

### scene_recognition.h
```c
#ifndef SCENE_RECOGNITION_H
#define SCENE_RECOGNITION_H

typedef enum {
    SCENE_MEETING = 0,      // 会议高潮
    SCENE_FOCUS = 1,        // 专注工作
    SCENE_PRESENTATION = 2, // 汇报演示
    SCENE_RELAX = 3,        // 休息放松
    SCENE_CLEANING = 4,     // 打扫卫生
} scene_id_t;

typedef struct {
    scene_id_t scene_id;
    float confidence;
    uint16_t color_temp;
    uint8_t brightness;
} scene_result_t;

int scene_init(void);
int scene_recognize(const float *radar_feat, const float *sound_feat,
                    scene_result_t *result);

#endif
```
